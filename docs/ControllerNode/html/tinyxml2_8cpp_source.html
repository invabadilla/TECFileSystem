276  const char* StrPair::GetStr()
277  {
278  TIXMLASSERT( _start );
279  TIXMLASSERT( _end );
280  if ( _flags & NEEDS_FLUSH ) {
281  *_end = 0;
282  _flags ^= NEEDS_FLUSH;
283 
284  if ( _flags ) {
285  const char* p = _start; // the read pointer
286  char* q = _start; // the write pointer
287 
288  while( p < _end ) {
289  if ( (_flags & NEEDS_NEWLINE_NORMALIZATION) && *p == CR ) {
290  // CR-LF pair becomes LF
291  // CR alone becomes LF
292  // LF-CR becomes LF
293  if ( *(p+1) == LF ) {
294  p += 2;
295  }
296  else {
297  ++p;
298  }
299  *q = LF;
300  ++q;
301  }
302  else if ( (_flags & NEEDS_NEWLINE_NORMALIZATION) && *p == LF ) {
303  if ( *(p+1) == CR ) {
304  p += 2;
305  }
306  else {
307  ++p;
308  }
309  *q = LF;
310  ++q;
311  }
312  else if ( (_flags & NEEDS_ENTITY_PROCESSING) && *p == '&' ) {
313  // Entities handled by tinyXML2:
314  // - special entities in the entity table [in/out]
315  // - numeric character reference [in]
316  // &#20013; or &#x4e2d;
317 
318  if ( *(p+1) == '#' ) {
319  const int buflen = 10;
320  char buf[buflen] = { 0 };
321  int len = 0;
322  const char* adjusted = const_cast<char*>( XMLUtil::GetCharacterRef( p, buf, &len ) );
323  if ( adjusted == 0 ) {
324  *q = *p;
325  ++p;
326  ++q;
327  }
328  else {
329  TIXMLASSERT( 0 <= len && len <= buflen );
330  TIXMLASSERT( q + len <= adjusted );
331  p = adjusted;
332  memcpy( q, buf, len );
333  q += len;
334  }
335  }
336  else {
337  bool entityFound = false;
338  for( int i = 0; i < NUM_ENTITIES; ++i ) {
339  const Entity& entity = entities[i];
340  if ( strncmp( p + 1, entity.pattern, entity.length ) == 0
341  && *( p + entity.length + 1 ) == ';' ) {
342  // Found an entity - convert.
343  *q = entity.value;
344  ++q;
345  p += entity.length + 2;
346  entityFound = true;
347  break;
348  }
349  }
350  if ( !entityFound ) {
351  // fixme: treat as error?
352  ++p;
353  ++q;
354  }
355  }
356  }
357  else {
358  *q = *p;
359  ++p;
360  ++q;
361  }
362  }
363  *q = 0;
364  }
365  // The loop below has plenty going on, and this
366  // is a less useful mode. Break it out.
367  if ( _flags & NEEDS_WHITESPACE_COLLAPSING ) {
368  CollapseWhitespace();
369  }
370  _flags = (_flags & NEEDS_DELETE);
371  }
372  TIXMLASSERT( _start );
373  return _start;
374  }

412  void XMLUtil::ConvertUTF32ToUTF8( unsigned long input, char* output, int* length )
413  {
414  const unsigned long BYTE_MASK = 0xBF;
415  const unsigned long BYTE_MARK = 0x80;
416  const unsigned long FIRST_BYTE_MARK[7] = { 0x00, 0x00, 0xC0, 0xE0, 0xF0, 0xF8, 0xFC };
417 
418  if (input < 0x80) {
419  *length = 1;
420  }
421  else if ( input < 0x800 ) {
422  *length = 2;
423  }
424  else if ( input < 0x10000 ) {
425  *length = 3;
426  }
427  else if ( input < 0x200000 ) {
428  *length = 4;
429  }
430  else {
431  *length = 0; // This code won't convert this correctly anyway.
432  return;
433  }
434 
435  output += *length;
436 
437  // Scary scary fall throughs are annotated with carefully designed comments
438  // to suppress compiler warnings such as -Wimplicit-fallthrough in gcc
439  switch (*length) {
440  case 4:
441  --output;
442  *output = static_cast<char>((input | BYTE_MARK) & BYTE_MASK);
443  input >>= 6;
444  //fall through
445  case 3:
446  --output;
447  *output = static_cast<char>((input | BYTE_MARK) & BYTE_MASK);
448  input >>= 6;
449  //fall through
450  case 2:
451  --output;
452  *output = static_cast<char>((input | BYTE_MARK) & BYTE_MASK);
453  input >>= 6;
454  //fall through
455  case 1:
456  --output;
457  *output = static_cast<char>(input | FIRST_BYTE_MARK[*length]);
458  break;
459  default:
460  TIXMLASSERT( false );
461  }
462  }

465  const char* XMLUtil::GetCharacterRef( const char* p, char* value, int* length )
466  {
467  // Presume an entity, and pull it out.
468  *length = 0;
469 
470  if ( *(p+1) == '#' && *(p+2) ) {
471  unsigned long ucs = 0;
472  TIXMLASSERT( sizeof( ucs ) >= 4 );
473  ptrdiff_t delta = 0;
474  unsigned mult = 1;
475  static const char SEMICOLON = ';';
476 
477  if ( *(p+2) == 'x' ) {
478  // Hexadecimal.
479  const char* q = p+3;
480  if ( !(*q) ) {
481  return 0;
482  }
483 
484  q = strchr( q, SEMICOLON );
485 
486  if ( !q ) {
487  return 0;
488  }
489  TIXMLASSERT( *q == SEMICOLON );
490 
491  delta = q-p;
492  --q;
493 
494  while ( *q != 'x' ) {
495  unsigned int digit = 0;
496 
497  if ( *q >= '0' && *q <= '9' ) {
498  digit = *q - '0';
499  }
500  else if ( *q >= 'a' && *q <= 'f' ) {
501  digit = *q - 'a' + 10;
502  }
503  else if ( *q >= 'A' && *q <= 'F' ) {
504  digit = *q - 'A' + 10;
505  }
506  else {
507  return 0;
508  }
509  TIXMLASSERT( digit < 16 );
510  TIXMLASSERT( digit == 0 || mult <= UINT_MAX / digit );
511  const unsigned int digitScaled = mult * digit;
512  TIXMLASSERT( ucs <= ULONG_MAX - digitScaled );
513  ucs += digitScaled;
514  TIXMLASSERT( mult <= UINT_MAX / 16 );
515  mult *= 16;
516  --q;
517  }
518  }
519  else {
520  // Decimal.
521  const char* q = p+2;
522  if ( !(*q) ) {
523  return 0;
524  }
525 
526  q = strchr( q, SEMICOLON );
527 
528  if ( !q ) {
529  return 0;
530  }
531  TIXMLASSERT( *q == SEMICOLON );
532 
533  delta = q-p;
534  --q;
535 
536  while ( *q != '#' ) {
537  if ( *q >= '0' && *q <= '9' ) {
538  const unsigned int digit = *q - '0';
539  TIXMLASSERT( digit < 10 );
540  TIXMLASSERT( digit == 0 || mult <= UINT_MAX / digit );
541  const unsigned int digitScaled = mult * digit;
542  TIXMLASSERT( ucs <= ULONG_MAX - digitScaled );
543  ucs += digitScaled;
544  }
545  else {
546  return 0;
547  }
548  TIXMLASSERT( mult <= UINT_MAX / 10 );
549  mult *= 10;
550  --q;
551  }
552  }
553  // convert the UCS to UTF-8
554  ConvertUTF32ToUTF8( ucs, value, length );
555  return p + delta + 1;
556  }
557  return p+1;
558  }

1037  char* XMLNode::ParseDeep( char* p, StrPair* parentEndTag, int* curLineNumPtr )
1038  {
1039  // This is a recursive method, but thinking about it "at the current level"
1040  // it is a pretty simple flat list:
1041  // <foo/>
1042  // <!-- comment -->
1043  //
1044  // With a special case:
1045  // <foo>
1046  // </foo>
1047  // <!-- comment -->
1048  //
1049  // Where the closing element (/foo) *must* be the next thing after the opening
1050  // element, and the names must match. BUT the tricky bit is that the closing
1051  // element will be read by the child.
1052  //
1053  // 'endTag' is the end tag for this node, it is returned by a call to a child.
1054  // 'parentEnd' is the end tag for the parent, which is filled in and returned.
1055 
1056  XMLDocument::DepthTracker tracker(_document);
1057  if (_document->Error())
1058  return 0;
1059 
1060  while( p && *p ) {
1061  XMLNode* node = 0;
1062 
1063  p = _document->Identify( p, &node );
1064  TIXMLASSERT( p );
1065  if ( node == 0 ) {
1066  break;
1067  }
1068 
1069  const int initialLineNum = node->_parseLineNum;
1070 
1071  StrPair endTag;
1072  p = node->ParseDeep( p, &endTag, curLineNumPtr );
1073  if ( !p ) {
1074  DeleteNode( node );
1075  if ( !_document->Error() ) {
1076  _document->SetError( XML_ERROR_PARSING, initialLineNum, 0);
1077  }
1078  break;
1079  }
1080 
1081  const XMLDeclaration* const decl = node->ToDeclaration();
1082  if ( decl ) {
1083  // Declarations are only allowed at document level
1084  //
1085  // Multiple declarations are allowed but all declarations
1086  // must occur before anything else.
1087  //
1088  // Optimized due to a security test case. If the first node is
1089  // a declaration, and the last node is a declaration, then only
1090  // declarations have so far been added.
1091  bool wellLocated = false;
1092 
1093  if (ToDocument()) {
1094  if (FirstChild()) {
1095  wellLocated =
1096  FirstChild() &&
1097  FirstChild()->ToDeclaration() &&
1098  LastChild() &&
1099  LastChild()->ToDeclaration();
1100  }
1101  else {
1102  wellLocated = true;
1103  }
1104  }
1105  if ( !wellLocated ) {
1106  _document->SetError( XML_ERROR_PARSING_DECLARATION, initialLineNum, "XMLDeclaration value=%s", decl->Value());
1107  DeleteNode( node );
1108  break;
1109  }
1110  }
1111 
1112  XMLElement* ele = node->ToElement();
1113  if ( ele ) {
1114  // We read the end tag. Return it to the parent.
1115  if ( ele->ClosingType() == XMLElement::CLOSING ) {
1116  if ( parentEndTag ) {
1117  ele->_value.TransferTo( parentEndTag );
1118  }
1119  node->_memPool->SetTracked(); // created and then immediately deleted.
1120  DeleteNode( node );
1121  return p;
1122  }
1123 
1124  // Handle an end tag returned to this level.
1125  // And handle a bunch of annoying errors.
1126  bool mismatch = false;
1127  if ( endTag.Empty() ) {
1128  if ( ele->ClosingType() == XMLElement::OPEN ) {
1129  mismatch = true;
1130  }
1131  }
1132  else {
1133  if ( ele->ClosingType() != XMLElement::OPEN ) {
1134  mismatch = true;
1135  }
1136  else if ( !XMLUtil::StringEqual( endTag.GetStr(), ele->Name() ) ) {
1137  mismatch = true;
1138  }
1139  }
1140  if ( mismatch ) {
1141  _document->SetError( XML_ERROR_MISMATCHED_ELEMENT, initialLineNum, "XMLElement name=%s", ele->Name());
1142  DeleteNode( node );
1143  break;
1144  }
1145  }
1146  InsertEndChild( node );
1147  }
1148  return 0;
1149  }

1922  char* XMLElement::ParseAttributes( char* p, int* curLineNumPtr )
1923  {
1924  XMLAttribute* prevAttribute = 0;
1925 
1926  // Read the attributes.
1927  while( p ) {
1928  p = XMLUtil::SkipWhiteSpace( p, curLineNumPtr );
1929  if ( !(*p) ) {
1930  _document->SetError( XML_ERROR_PARSING_ELEMENT, _parseLineNum, "XMLElement name=%s", Name() );
1931  return 0;
1932  }
1933 
1934  // attribute.
1935  if (XMLUtil::IsNameStartChar( (unsigned char) *p ) ) {
1936  XMLAttribute* attrib = CreateAttribute();
1937  TIXMLASSERT( attrib );
1938  attrib->_parseLineNum = _document->_parseCurLineNum;
1939 
1940  const int attrLineNum = attrib->_parseLineNum;
1941 
1942  p = attrib->ParseDeep( p, _document->ProcessEntities(), curLineNumPtr );
1943  if ( !p || Attribute( attrib->Name() ) ) {
1944  DeleteAttribute( attrib );
1945  _document->SetError( XML_ERROR_PARSING_ATTRIBUTE, attrLineNum, "XMLElement name=%s", Name() );
1946  return 0;
1947  }
1948  // There is a minor bug here: if the attribute in the source xml
1949  // document is duplicated, it will not be detected and the
1950  // attribute will be doubly added. However, tracking the 'prevAttribute'
1951  // avoids re-scanning the attribute list. Preferring performance for
1952  // now, may reconsider in the future.
1953  if ( prevAttribute ) {
1954  TIXMLASSERT( prevAttribute->_next == 0 );
1955  prevAttribute->_next = attrib;
1956  }
1957  else {
1958  TIXMLASSERT( _rootAttribute == 0 );
1959  _rootAttribute = attrib;
1960  }
1961  prevAttribute = attrib;
1962  }
1963  // end of the tag
1964  else if ( *p == '>' ) {
1965  ++p;
1966  break;
1967  }
1968  // end of the tag
1969  else if ( *p == '/' && *(p+1) == '>' ) {
1970  _closingType = CLOSED;
1971  return p+2; // done; sealed element.
1972  }
1973  else {
1974  _document->SetError( XML_ERROR_PARSING_ELEMENT, _parseLineNum, 0 );
1975  return 0;
1976  }
1977  }
1978  return p;
1979  }
